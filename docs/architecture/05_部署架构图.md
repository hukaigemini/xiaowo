# 小窝项目部署架构图

## 部署架构概述

小窝项目采用现代化的容器化部署架构，支持快速部署、水平扩展和高可用性。部署架构设计遵循DevOps最佳实践，确保系统的稳定性、安全性和可维护性。

## 部署环境配置

### 1. 开发环境 (Development)

```yaml
# docker-compose.dev.yml
version: '3.8'
services:
  nginx:
    image: nginx:alpine
    ports:
      - "80:80"
      - "443:443"
    volumes:
      - ./nginx/nginx.dev.conf:/etc/nginx/nginx.conf
      - ./frontend:/var/www/html
    networks:
      - xiaowo-net

  backend:
    build:
      context: ./backend
      dockerfile: Dockerfile.dev
    ports:
      - "8080:8080"
    environment:
      - ENV=development
      - DB_PATH=/app/data/dev.db
      - REDIS_URL=redis://redis:6379
    volumes:
      - ./backend:/app
      - ./data:/app/data
    networks:
      - xiaowo-net
    depends_on:
      - redis

  redis:
    image: redis:alpine
    ports:
      - "6379:6379"
    volumes:
      - redis-data:/data
    networks:
      - xiaowo-net

  # 监控
  prometheus:
    image: prom/prometheus
    ports:
      - "9090:9090"
    volumes:
      - ./monitoring/prometheus.yml:/etc/prometheus/prometheus.yml
    networks:
      - xiaowo-net

networks:
  xiaowo-net:
    driver: bridge

volumes:
  redis-data:
```

### 2. 防火墙规则

```bash
#!/bin/bash
# 防火墙配置脚本

# 清理现有规则
iptables -F
iptables -X
iptables -t nat -F
iptables -t nat -X

# 默认策略
iptables -P INPUT DROP
iptables -P FORWARD DROP
iptables -P OUTPUT ACCEPT

# 允许回环
iptables -A INPUT -i lo -j ACCEPT
iptables -A OUTPUT -o lo -j ACCEPT

# 允许已建立的连接
iptables -A INPUT -m conntrack --ctstate ESTABLISHED,RELATED -j ACCEPT

# 允许SSH (限制频率)
iptables -A INPUT -p tcp --dport 22 -m conntrack --ctstate NEW -m recent --set --name SSH
iptables -A INPUT -p tcp --dport 22 -m conntrack --ctstate NEW -m recent --update --seconds 60 --hitcount 4 --name SSH -j DROP
iptables -A INPUT -p tcp --dport 22 -j ACCEPT

# 允许HTTP/HTTPS
iptables -A INPUT -p tcp --dport 80 -j ACCEPT
iptables -A INPUT -p tcp --dport 443 -j ACCEPT

# 允许WebSocket
iptables -A INPUT -p tcp --dport 8080 -j ACCEPT

# 允许监控端口
iptables -A INPUT -p tcp --dport 9090 -j ACCEPT  # Prometheus
iptables -A INPUT -p tcp --dport 3000 -j ACCEPT  # Grafana

# 保存规则
iptables-save > /etc/iptables/rules.v4
```

## 安全配置

### 1. SSL/TLS配置

```nginx
# nginx.prod.conf
server {
    listen 443 ssl http2;
    server_name xiaowo.example.com;

    # SSL配置
    ssl_certificate /etc/ssl/certs/xiaowo.crt;
    ssl_certificate_key /etc/ssl/private/xiaowo.key;
    ssl_protocols TLSv1.2 TLSv1.3;
    ssl_ciphers ECDHE-RSA-AES256-GCM-SHA512:DHE-RSA-AES256-GCM-SHA512;
    ssl_prefer_server_ciphers off;
    ssl_session_cache shared:SSL:10m;
    ssl_session_timeout 10m;

    # 安全头
    add_header Strict-Transport-Security "max-age=63072000; includeSubDomains; preload";
    add_header X-Frame-Options DENY;
    add_header X-Content-Type-Options nosniff;
    add_header X-XSS-Protection "1; mode=block";
    add_header Content-Security-Policy "default-src 'self'; script-src 'self' 'unsafe-inline'; style-src 'self' 'unsafe-inline';";

    # 负载均衡配置
    upstream backend {
        least_conn;
        server backend1:8080 weight=1 max_fails=3 fail_timeout=30s;
        server backend2:8080 weight=1 max_fails=3 fail_timeout=30s;
        server backend3:8080 weight=1 max_fails=3 fail_timeout=30s;
    }

    # WebSocket代理
    location /ws/ {
        proxy_pass http://backend;
        proxy_http_version 1.1;
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection "upgrade";
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
```

## 备份策略

### 1. 数据备份

```bash
#!/bin/bash
# backup.sh - 数据库备份脚本

DB_NAME="xiaowo_production"
DB_USER="backup_user"
DB_PASS="backup_password"
BACKUP_DIR="/backup/mysql"
DATE=$(date +%Y%m%d_%H%M%S)

# 创建备份目录
mkdir -p $BACKUP_DIR

# 数据库备份
mysqldump -u$DB_USER -p$DB_PASS \
  --single-transaction \
  --routines \
  --triggers \
  $DB_NAME > $BACKUP_DIR/xiaowo_backup_$DATE.sql

# 压缩备份文件
gzip $BACKUP_DIR/xiaowo_backup_$DATE.sql

# 删除7天前的备份
find $BACKUP_DIR -name "xiaowo_backup_*.sql.gz" -mtime +7 -delete

echo "Backup completed: xiaowo_backup_$DATE.sql.gz"
```
